import { describe,it } from 'node:test';import assert from 'node:assert/strict';import { __test,OPERATIONAL_SOURCE_POLICY } from '../src/legacy/action-sources';
describe('Phase 20 operational Action source bridge',()=>{it('maps real workflow outcomes to Action states',()=>{assert.equal(__test.transition('start'),'in_progress');assert.equal(__test.transition('waiting'),'waiting');assert.equal(__test.transition('complete'),'completed');assert.equal(__test.transition('reject'),'ready');assert.equal(__test.transition('cancel'),'cancelled');});it('does not reopen terminal Actions',()=>{assert.equal(__test.validSourceTransition('completed','start'),false);assert.equal(__test.validSourceTransition('cancelled','waiting'),false);assert.equal(__test.validSourceTransition('completed','complete'),true);});it('registers every Group 2 source family',()=>{for(const source of ['approval_request','scheduled_stock_take','stock_take_recount','incoming_transfer','expected_purchase_order','goods_receiving','wastage_exception','manufacturing_requirement','low_stock_intervention','interrupted_draft','unknown_barcode'])assert.ok(OPERATIONAL_SOURCE_POLICY[source],source);});});
