async function member(env:Env,w:string,userId:string){return env.CENTRAL_DB.prepare(`SELECT auth_uid,display_name,email,role,status FROM workspace_members WHERE workspace_id=?1 AND auth_uid=?2 LIMIT 1`).bind(w,userId).first<Row>();}
async function permitted(env:Env,w:string,userId:string,locationId:string){if(!locationId)return true;try{const row=await env.CENTRAL_DB.prepare(`SELECT 1 ok FROM workspace_member_locations WHERE workspace_id=?1 AND auth_uid=?2 AND location_id=?3 LIMIT 1`).bind(w,userId,locationId).first<Row>();return !!row;}catch{return false;}}
async function eligible(env:Env,w:string,userId:string,locationId:string,requiredPermission=''){const user=await member(env,w,userId);if(!user||t(user.status)!=='active')return {ok:false,message:'User is inactive.'};if(!(await permitted(env,w,userId,locationId)))return {ok:false,message:'Location is outside the user’s existing access.'};if(requiredPermission){try{const row=await env.CENTRAL_DB.prepare(`SELECT 1 ok FROM workspace_member_permissions WHERE workspace_id=?1 AND auth_uid=?2 AND permission=?3 LIMIT 1`).bind(w,userId,requiredPermission).first<Row>();if(!row)return {ok:false,message:`User lacks ${requiredPermission} permission.`};}catch{/* permission service remains authoritative at execution time */}}return {ok:true,message:''};}
