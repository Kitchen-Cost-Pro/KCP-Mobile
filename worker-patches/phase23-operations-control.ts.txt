import type { AuthContext, Env } from './types';
import { error, json, readJson } from './http';
import { assertWorkspaceAccess, assertWorkspacePermission, assertLocationAccess } from './auth';

type Row = Record<string, any>;
const text=(v:unknown,f='')=>String(v??f).trim(); const iso=()=>new Date().toISOString();
const managerRoles=new Set(['owner','admin','superuser','super-user','manager']);
const views=new Set(['all','location','role_set','user','overdue','waiting','blocked','completed','high_impact','unassigned']);
const priorities=new Set(['low','normal','high','urgent']);

export async function ensureOperationsControlSchema(env: Env) {
  for (const sql of [
    `CREATE TABLE IF NOT EXISTS action_management_events (id TEXT PRIMARY KEY, workspace_id TEXT NOT NULL, action_id TEXT NOT NULL, event_type TEXT NOT NULL, actor_uid TEXT NOT NULL, reason TEXT NOT NULL, detail_json TEXT NOT NULL DEFAULT '{}', created_at TEXT NOT NULL)`,
    `CREATE INDEX IF NOT EXISTS idx_action_management_events_action ON action_management_events(workspace_id,action_id,created_at)`,
    `CREATE INDEX IF NOT EXISTS idx_action_management_events_type ON action_management_events(workspace_id,event_type,created_at)`
  ]) await env.DB.prepare(sql).run();
}
async function assertManager(env: Env, auth: AuthContext, workspaceId: string) {
  await assertWorkspaceAccess(env,auth,workspaceId); await assertWorkspacePermission(env,auth,workspaceId,'nav-tasks');
  const member=await env.CENTRAL_DB.prepare(`SELECT role FROM workspace_members WHERE workspace_id=?1 AND auth_uid=?2 AND status='active'`).bind(workspaceId,auth.uid).first<Row>();
  if(!managerRoles.has(text(member?.role).toLowerCase().replace(/[_\s]+/g,'-'))) throw new Error('Manager access is required for Operations Control.');
}
async function managerLocationAccess(env: Env, auth: AuthContext, workspaceId: string, locationId: string) { if(locationId) await assertLocationAccess(env,auth,workspaceId,locationId,'operations_control'); }
function viewWhere(view:string) { switch(view) { case 'overdue': return ` AND i.status!='completed' AND i.status!='cancelled' AND i.due_at<?2`; case 'waiting': return ` AND COALESCE(m.status,i.status)='waiting'`; case 'blocked': return ` AND EXISTS (SELECT 1 FROM routine_policies p WHERE p.routine_id=i.template_id AND p.workspace_id=i.workspace_id AND p.dependencies_json!='[]')`; case 'completed': return ` AND i.status='completed'`; case 'high_impact': return ` AND COALESCE(m.financial_impact_amount,0)>0`; case 'unassigned': return ` AND (i.assignment_id='' OR i.assignment_label='' OR i.assignment_type='')`; default:return ''; } }
async function actions(env: Env, workspaceId: string, view: string, locationId: string) {
  const base=`SELECT i.*,m.action_type,m.source_type,m.source_id,m.status meta_status,m.required_permission,m.financial_impact_amount,m.currency,m.deep_link,m.progress_percent,l.name location_name FROM task_instances i LEFT JOIN action_metadata m ON m.workspace_id=i.workspace_id AND m.action_id=i.id LEFT JOIN locations l ON l.workspace_id=i.workspace_id AND l.id=i.location_id WHERE i.workspace_id=?1 ${locationId?'AND i.location_id=?3':''}${viewWhere(view)} ORDER BY CASE i.priority WHEN 'urgent' THEN 4 WHEN 'high' THEN 3 WHEN 'normal' THEN 2 ELSE 1 END DESC, i.due_at ASC LIMIT 500`;
  const result=await env.DB.prepare(base).bind(workspaceId,iso(),locationId).all<Row>(); return result.results||[];
}
async function safeAction(env: Env, workspaceId:string,row:Row) {
  const [steps,evidence,managerEvents,routingEvents]=await Promise.all([
    env.DB.prepare(`SELECT * FROM task_instance_steps WHERE workspace_id=?1 AND task_id=?2 ORDER BY sort_order`).bind(workspaceId,row.id).all<Row>(),
    env.DB.prepare(`SELECT id,step_id,file_name,mime_type,byte_size,created_by,created_at FROM task_evidence WHERE workspace_id=?1 AND task_id=?2 ORDER BY created_at`).bind(workspaceId,row.id).all<Row>(),
    env.DB.prepare(`SELECT * FROM action_management_events WHERE workspace_id=?1 AND action_id=?2 ORDER BY created_at DESC LIMIT 100`).bind(workspaceId,row.id).all<Row>(),
    env.DB.prepare(`SELECT * FROM action_routing_history WHERE workspace_id=?1 AND action_id=?2 ORDER BY created_at DESC LIMIT 100`).bind(workspaceId,row.id).all<Row>()
  ]);
  const memberRows=await env.CENTRAL_DB.prepare(`SELECT auth_uid,display_name,email FROM workspace_members WHERE workspace_id=?1`).bind(workspaceId).all<Row>(); const names=new Map((memberRows.results||[]).map(x=>[text(x.auth_uid),text(x.display_name||x.email||'KCP user')]));
  const currentSteps=(steps.results||[]).map(s=>({id:text(s.id),label:text(s.label),required:Number(s.required)===1,completed:Number(s.completed)===1,completedAt:text(s.completed_at)||undefined,note:text(s.note)})); const complete=currentSteps.filter(s=>s.completed).length;
  const status=text(row.meta_status||row.status)==='open'?(complete?'in_progress':'ready'):text(row.meta_status||row.status); const due=new Date(text(row.due_at));
  const history=[...(managerEvents.results||[]).map(e=>({id:text(e.id),eventType:text(e.event_type),actorName:names.get(text(e.actor_uid))||'KCP manager',occurredAt:text(e.created_at),detail:text(e.reason)})),...(routingEvents.results||[]).map(e=>({id:text(e.id),eventType:text(e.event_type),actorName:names.get(text(e.actor_uid))||'KCP user',occurredAt:text(e.created_at),detail:text(e.reason)}))].sort((a,b)=>b.occurredAt.localeCompare(a.occurredAt));
  return {id:text(row.id),workspaceId,routineId:text(row.template_id),actionType:text(row.action_type,'routine_checklist'),sourceRecord:{type:text(row.source_type,'routine'),id:text(row.source_id,row.template_id)},title:text(row.title),description:text(row.description),priority:text(row.priority,'normal'),dueAt:text(row.due_at),status,location:row.location_id?{id:text(row.location_id),name:text(row.location_name,row.location_id)}:null,assignment:{type:text(row.assignment_type)==='role'?'role_set':text(row.assignment_type||'location'),id:text(row.assignment_id),label:text(row.assignment_label||'Unassigned')},progress:{completed:complete,total:currentSteps.length,percent:currentSteps.length?Math.round(complete*100/currentSteps.length):0},requiredPermission:text(row.required_permission,'nav-tasks'),financialImpact:row.financial_impact_amount==null?null:{amount:Number(row.financial_impact_amount),currency:text(row.currency,'ZAR')},deepLink:text(row.deep_link),steps:currentSteps,notes:text(row.notes),evidence:(evidence.results||[]).map(e=>({id:text(e.id),stepId:text(e.step_id)||null,fileName:text(e.file_name),mimeType:text(e.mime_type),byteSize:Number(e.byte_size),createdAt:text(e.created_at),createdByName:names.get(text(e.created_by))||'KCP user'})),activityHistory:history,revision:Number(row.revision||1),completedAt:text(row.completed_at)||undefined,completedByName:names.get(text(row.completed_by))||undefined,updatedAt:text(row.updated_at),isOverdue:status!=='completed'&&status!=='cancelled'&&!Number.isNaN(due.getTime())&&due.getTime()<Date.now()};
}
async function metrics(env:Env,workspaceId:string,rows:Row[]) { const cutoff=new Date(Date.now()-30*86400000).toISOString(); const completed=rows.filter(x=>text(x.status)==='completed'); const active=rows.filter(x=>!['completed','cancelled'].includes(text(x.status))); const overdue=active.filter(x=>new Date(text(x.due_at)).getTime()<Date.now()); const completionTimes=completed.map(x=>x.completed_at?Math.max(0,(new Date(x.completed_at).getTime()-new Date(x.created_at).getTime())/60000):null).filter((x):x is number=>x!=null); const deferred=await env.DB.prepare(`SELECT COUNT(*) count FROM action_management_events WHERE workspace_id=?1 AND event_type='deferred' AND created_at>=?2`).bind(workspaceId,cutoff).first<Row>(); const rejected=await env.DB.prepare(`SELECT COUNT(*) count FROM action_management_events WHERE workspace_id=?1 AND event_type='rejected' AND created_at>=?2`).bind(workspaceId,cutoff).first<Row>(); const resolved=await env.DB.prepare(`SELECT COALESCE(SUM(COALESCE(m.financial_impact_amount,0)),0) value,MAX(m.currency) currency FROM task_instances i LEFT JOIN action_metadata m ON m.workspace_id=i.workspace_id AND m.action_id=i.id WHERE i.workspace_id=?1 AND i.status='completed' AND i.completed_at>=?2`).bind(workspaceId,cutoff).first<Row>();
  const byLocation=new Map<string,{locationId:string;locationName:string;openActions:number;overdueActions:number}>(); const workload=new Map<string,{key:string;label:string;kind:'user'|'role_set';openActions:number;overdueActions:number}>(); for(const row of rows){const isOpen=!['completed','cancelled'].includes(text(row.status));const isOverdue=isOpen&&new Date(text(row.due_at)).getTime()<Date.now();if(row.location_id){const key=text(row.location_id);const item=byLocation.get(key)||{locationId:key,locationName:text(row.location_name,key),openActions:0,overdueActions:0};item.openActions+=Number(isOpen);item.overdueActions+=Number(isOverdue);byLocation.set(key,item);}if(isOpen&&['user','role'].includes(text(row.assignment_type))){const kind=text(row.assignment_type)==='role'?'role_set':'user';const key=`${kind}:${text(row.assignment_id)}`;const item=workload.get(key)||{key,label:text(row.assignment_label||row.assignment_id),kind,openActions:0,overdueActions:0};item.openActions++;item.overdueActions+=Number(isOverdue);workload.set(key,item);}}
  return {completionRate:rows.length?Math.round(completed.length*100/rows.length):0,averageCompletionMinutes:completionTimes.length?Math.round(completionTimes.reduce((a,b)=>a+b,0)/completionTimes.length):null,overdueRate:active.length?Math.round(overdue.length*100/active.length):0,approvalTurnaroundMinutes:null,deferredActions:Number(deferred?.count||0),rejectedSubmissions:Number(rejected?.count||0),financialImpactResolved:Number(resolved?.value||0),currency:text(resolved?.currency,'ZAR'),locationReadiness:[...byLocation.values()].map(x=>({...x,readiness:Math.max(0,Math.round(100-(x.overdueActions*100/Math.max(1,x.openActions))))})),workload:[...workload.values()]};
}
export async function getOperationsControl(request:Request,env:Env,auth:AuthContext,workspaceId:string){try{await assertManager(env,auth,workspaceId);await ensureOperationsControlSchema(env);const url=new URL(request.url);const view=text(url.searchParams.get('view'),'all');if(!views.has(view))return error(request,env,400,'Invalid Operations Control view.');const locationId=text(url.searchParams.get('locationId'));await managerLocationAccess(env,auth,workspaceId,locationId);const rows=await actions(env,workspaceId,view,locationId);const safe=await Promise.all(rows.map(row=>safeAction(env,workspaceId,row)));return json(request,env,{ok:true,generatedAt:iso(),actions:safe,metrics:await metrics(env,workspaceId,rows),filters:{view,locationId:locationId||undefined}});}catch(cause){return error(request,env,403,cause instanceof Error?cause.message:'Operations Control is unavailable.');}}
async function actionRow(env:Env,workspaceId:string,actionId:string){return env.DB.prepare(`SELECT i.*,m.status meta_status,m.financial_impact_amount,m.currency,m.deep_link,m.source_type,m.source_id,m.action_type,l.name location_name FROM task_instances i LEFT JOIN action_metadata m ON m.workspace_id=i.workspace_id AND m.action_id=i.id LEFT JOIN locations l ON l.workspace_id=i.workspace_id AND l.id=i.location_id WHERE i.workspace_id=?1 AND i.id=?2`).bind(workspaceId,actionId).first<Row>();}
async function validateAssignment(env:Env,workspaceId:string,type:string,targetId:string,locationId:string){if(!['user','role'].includes(type)||!targetId)throw new Error('Choose an eligible user or Role Set.');if(type==='user'){const user=await env.CENTRAL_DB.prepare(`SELECT auth_uid FROM workspace_members WHERE workspace_id=?1 AND auth_uid=?2 AND status='active'`).bind(workspaceId,targetId).first<Row>();if(!user)throw new Error('This user is not active in the workspace.');if(locationId){try{const membership=await env.CENTRAL_DB.prepare(`SELECT 1 ok FROM workspace_member_locations WHERE workspace_id=?1 AND auth_uid=?2 AND location_id=?3 LIMIT 1`).bind(workspaceId,targetId,locationId).first<Row>();if(!membership)throw new Error('This user is not authorised for this location.');}catch(cause){if(cause instanceof Error&&cause.message.includes('not authorised'))throw cause;}}}}
export async function postOperationsIntervention(request:Request,env:Env,auth:AuthContext,workspaceId:string,actionId:string,operation:string){try{await assertManager(env,auth,workspaceId);await ensureOperationsControlSchema(env);const body=await readJson<Row>(request);const reason=text(body.reason);if(!reason)return error(request,env,400,'A manager reason is required.');const row=await actionRow(env,workspaceId,actionId);if(!row)return error(request,env,404,'Action not found.');await managerLocationAccess(env,auth,workspaceId,text(row.location_id));const timestamp=iso();const eventType=operation==='priority'?'priority_changed':operation==='resolve_blocker'?'blocker_resolved':operation==='defer'?'deferred':operation==='escalate'?'escalated':'reassigned';const statements:any[]=[];if(operation==='reassign'){const type=text(body.type);const targetId=text(body.targetId);await validateAssignment(env,workspaceId,type,targetId,text(row.location_id));statements.push(env.DB.prepare(`UPDATE task_instances SET assignment_type=?1,assignment_id=?2,assignment_label=?3,revision=revision+1,updated_at=?4 WHERE workspace_id=?5 AND id=?6`).bind(type,targetId,text(body.label||targetId).slice(0,160),timestamp,workspaceId,actionId));}else if(operation==='priority'){const priority=text(body.priority);if(!priorities.has(priority))return error(request,env,400,'Invalid priority.');statements.push(env.DB.prepare(`UPDATE task_instances SET priority=?1,revision=revision+1,updated_at=?2 WHERE workspace_id=?3 AND id=?4`).bind(priority,timestamp,workspaceId,actionId));}else if(operation==='defer'){if(text(row.status)==='completed')return error(request,env,409,'Completed Actions cannot be deferred.');statements.push(env.DB.prepare(`UPDATE action_metadata SET status='deferred',updated_at=?1 WHERE workspace_id=?2 AND action_id=?3`).bind(timestamp,workspaceId,actionId));}else if(operation==='escalate'){statements.push(env.DB.prepare(`UPDATE action_metadata SET status=CASE WHEN status='completed' THEN status ELSE 'waiting' END,updated_at=?1 WHERE workspace_id=?2 AND action_id=?3`).bind(timestamp,workspaceId,actionId));}else if(operation!=='resolve_blocker')return error(request,env,400,'Unsupported manager operation.');statements.push(env.DB.prepare(`INSERT INTO action_management_events (id,workspace_id,action_id,event_type,actor_uid,reason,detail_json,created_at) VALUES (?1,?2,?3,?4,?5,?6,?7,?8)`).bind(`action-manager-${crypto.randomUUID()}`,workspaceId,actionId,eventType,auth.uid,reason.slice(0,1000),JSON.stringify(body),timestamp));await env.DB.batch(statements);const updated=await actionRow(env,workspaceId,actionId);return json(request,env,{ok:true,action:await safeAction(env,workspaceId,updated||row)});}catch(cause){return error(request,env,403,cause instanceof Error?cause.message:'Manager intervention failed.');}}
